2. Validar "nombre_producto" en el Insert
Tu tabla Detalle_Venta pide la columna nombre_producto. Debes agregarla al comando SQL de inserción dentro del método btnFinalizar_Click.

Busca el paso 2 dentro de btnFinalizar_Click y actualiza el comando así:

// 2. Recorrer el DataGridView para insertar detalles y descontar stock
foreach (DataGridViewRow fila in dgvCarrito.Rows)
{
    if (fila.Cells[0].Value == null) continue;

    int idProd = Convert.ToInt32(fila.Cells[0].Value);
    string nomProd = fila.Cells[1].Value.ToString(); // Obtener el nombre de la celda
    int cant = Convert.ToInt32(fila.Cells[2].Value);
    double precio = Convert.ToDouble(fila.Cells[3].Value);

    // INSERT ACTUALIZADO (Añadido nombre_producto)
    string sqlDetalle = "INSERT INTO Detalle_Venta (id_venta, id_producto, nombre_producto, cantidad, precio) VALUES (@idV, @idP, @nom, @cant, @pre)";
    SQLiteCommand cmdDetalle = new SQLiteCommand(sqlDetalle, conexion, transaccion);
    cmdDetalle.Parameters.AddWithValue("@idV", idVenta);
    cmdDetalle.Parameters.AddWithValue("@idP", idProd);
    cmdDetalle.Parameters.AddWithValue("@nom", nomProd); // Pasamos el nombre
    cmdDetalle.Parameters.AddWithValue("@cant", cant);
    cmdDetalle.Parameters.AddWithValue("@pre", precio);
    cmdDetalle.ExecuteNonQuery();

    // ... (El código de actualizar stock sigue igual) ...
}
